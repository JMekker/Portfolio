	let count = 0;		// This is to alternate between which object to change.
	let multiplier = 225;	// Change this to alter the maximum allowable contrast.
	let adder = 30;		// Change this to alter the minimum allowable contrast.
				// Make sure these two numbers, added up, do not exceed 255.

/*
Note: I adjusted the values for this page specifically so that it's more readable. The multiplier and adder values on this page are 130 and 125 respectively.
*/ 


	function changeAllColors() {	// This changes all of the colors to some (kinda) random colors
		let x1 = Math.floor(Math.random() * multiplier + adder);
		let x2 = Math.floor(Math.random() * multiplier + adder);
		let x3 = Math.floor(Math.random() * multiplier + adder);
		let background = rgbToHex(x1, x2, x3);

		let y1 = Math.floor(Math.random() * multiplier);
		let y2 = Math.floor(Math.random() * multiplier);
		let y3 = Math.floor(Math.random() * multiplier);
		let text = rgbToHex(y1, y2, y3);

		let z1 = Math.floor(Math.random() * multiplier + adder);
		let z2 = Math.floor(Math.random() * multiplier + adder);
		let z3 = Math.floor(Math.random() * multiplier + adder);
		let glow = rgbToHex(z1, z2, z3);

		document.querySelector(':root').style.setProperty('--background', background);
		document.querySelector(':root').style.setProperty('--text', text);
		document.querySelector(':root').style.setProperty('--glow', glow);
	}

	function changeColors() {	//This changes only one object at a time, alternating between them.
		if (count == 0) {
			let x1 = Math.floor(Math.random() * multiplier + adder);
			let x2 = Math.floor(Math.random() * multiplier + adder);
			let x3 = Math.floor(Math.random() * multiplier + adder);
			let background = rgbToHex(x1, x2, x3);

			document.querySelector(':root').style.setProperty('--background', background);
			count++;
		} else if (count == 1) {
			let y1 = Math.floor(Math.random() * multiplier);
			let y2 = Math.floor(Math.random() * multiplier);
			let y3 = Math.floor(Math.random() * multiplier);
			let text = rgbToHex(y1, y2, y3);

			document.querySelector(':root').style.setProperty('--text', text);
			count++;
		} else {
			let z1 = Math.floor(Math.random() * multiplier + adder);
			let z2 = Math.floor(Math.random() * multiplier + adder);
			let z3 = Math.floor(Math.random() * multiplier + adder);
			let glow = rgbToHex(z1, z2, z3);

			document.querySelector(':root').style.setProperty('--glow', glow);
			count = 0;
		}
	}

	function rgbToHex(r, g, b) {	// This changes the RGB(#, #, #) that we generate into a hex (like #aabbcc)
		return "#" + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);
	} 

	window.onload=function () {	// This changes all of the colors as soon as the page loads.
		changeAllColors();
	};
